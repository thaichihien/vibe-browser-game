// ===== RULES-START =====
const Rules = (() => {
  const ROWS = 9, COLS = 7;
  const RANKS = { CHUOT:1, MEO:2, CHO:3, SOI:4, BAO:5, HO:6, SUTU:7, VOI:8 };
  const NAMES = { 1:'Chu·ªôt', 2:'M√®o', 3:'Ch√≥', 4:'Ch√≥ s√≥i', 5:'B√°o', 6:'H·ªï', 7:'S∆∞ t·ª≠', 8:'Voi' };
  function terrain(r, c) {
    if (r >= 3 && r <= 5 && (c === 1 || c === 2 || c === 4 || c === 5)) return 'water';
    if (r === 0 && c === 3) return 'denB';
    if (r === 8 && c === 3) return 'denR';
    if ((r === 0 && (c === 2 || c === 4)) || (r === 1 && c === 3)) return 'trapB';
    if ((r === 8 && (c === 2 || c === 4)) || (r === 7 && c === 3)) return 'trapR';
    return 'land';
  }
  // Blue top rows 0-2; Red is the 180¬∞ mirror: (r,c) -> (8-r, 6-c)
  const BLUE_START = [[0,0,7],[0,6,6],[1,1,3],[1,5,2],[2,0,1],[2,2,5],[2,4,4],[2,6,8]]; // [r,c,rank]
  function initialState() {
    const board = Array.from({ length: ROWS }, () => Array(COLS).fill(null));
    for (const [r, c, rank] of BLUE_START) {
      board[r][c] = { side:'B', rank, id:'B' + rank };
      board[8 - r][6 - c] = { side:'R', rank, id:'R' + rank };
    }
    return { board, turn:'R', winner:null, reason:null };
  }
  return { ROWS, COLS, RANKS, NAMES, terrain, initialState };
})();
// ===== RULES-END =====

```js
const inBounds = (r, c) => r >= 0 && r < ROWS && c >= 0 && c < COLS;
const isWater = (r, c) => terrain(r, c) === 'water';
function effectiveRank(state, r, c) {
  const p = state.board[r][c]; if (!p) return 0;
  const t = terrain(r, c);
  if ((t === 'trapR' && p.side === 'B') || (t === 'trapB' && p.side === 'R')) return 0;
  return p.rank;
}
function canCapture(state, fr, fc, tr, tc) {
  const a = state.board[fr][fc], d = state.board[tr][tc];
  if (!d || d.side === a.side) return false;
  if (isWater(fr, fc) !== isWater(tr, tc)) return false;      // no cross-boundary
  const dr = effectiveRank(state, tr, tc);
  if (a.rank === RANKS.CHUOT && d.rank === RANKS.VOI) return true;
  if (a.rank === RANKS.VOI && d.rank === RANKS.CHUOT && dr !== 0) return false;
  return a.rank >= dr;
}
const JUMP_AXES = { 7: 'both', 6: 'vertical', 5: 'horizontal' };
function jumpTargets(state, r, c) {
  const p = state.board[r][c], axes = JUMP_AXES[p.rank];
  if (!axes) return [];
  const out = [];
  const dirs = [];
  if (axes === 'both' || axes === 'vertical') dirs.push([1, 0], [-1, 0]);
  if (axes === 'both' || axes === 'horizontal') dirs.push([0, 1], [0, -1]);
  for (const [dr, dc] of dirs) {
    let rr = r + dr, cc = c + dc;
    if (!inBounds(rr, cc) || !isWater(rr, cc)) continue; // must start jumping over water
    let blocked = false;
    while (inBounds(rr, cc) && isWater(rr, cc)) {
      if (state.board[rr][cc]) blocked = true;            // any piece in water = a rat = blocks
      rr += dr; cc += dc;
    }
    if (blocked || !inBounds(rr, cc)) continue;
    out.push([rr, cc]);
  }
  return out;
}
function legalMoves(state, r, c) {
  const p = state.board[r][c];
  if (!p || p.side !== state.turn) return [];
  const moves = [];
  const consider = (tr, tc, jump) => {
    const t = terrain(tr, tc);
    if ((t === 'denR' && p.side === 'R') || (t === 'denB' && p.side === 'B')) return;
    if (isWater(tr, tc) && p.rank !== RANKS.CHUOT) return;
    const target = state.board[tr][tc];
    if (target) {
      if (canCapture(state, r, c, tr, tc)) moves.push({ r: tr, c: tc, capture: true, jump });
    } else moves.push({ r: tr, c: tc, capture: false, jump });
  };
  for (const [dr, dc] of [[1,0],[-1,0],[0,1],[0,-1]]) {
    const tr = r + dr, tc = c + dc;
    if (inBounds(tr, tc)) consider(tr, tc, false);
  }
  for (const [tr, tc] of jumpTargets(state, r, c)) consider(tr, tc, true);
  return moves;
}
function allMoves(state, side) {
  const out = [];
  for (let r = 0; r < ROWS; r++) for (let c = 0; c < COLS; c++) {
    const p = state.board[r][c];
    if (p && p.side === side) for (const m of legalMoves({ ...state, turn: side }, r, c)) out.push({ from: { r, c }, to: m });
  }
  return out;
}
```

function countPieces(state, side) {
  return state.board.flat().filter(p => p && p.side === side).length;
}
function applyMove(state, from, to) {
  const ok = legalMoves(state, from.r, from.c).some(m => m.r === to.r && m.c === to.c);
  if (!ok) throw new Error('illegal move');
  const piece = state.board[from.r][from.c];
  const captured = state.board[to.r][to.c];
  state.board[to.r][to.c] = piece; state.board[from.r][from.c] = null;
  const enemy = piece.side === 'R' ? 'B' : 'R';
  let winner = null, reason = null;
  const t = terrain(to.r, to.c);
  if ((t === 'denB' && piece.side === 'R') || (t === 'denR' && piece.side === 'B')) { winner = piece.side; reason = 'den'; }
  else if (countPieces(state, enemy) === 0) { winner = piece.side; reason = 'annihilation'; }
  else if (allMoves(state, enemy).length === 0) { winner = piece.side; reason = 'stalemate'; }
  state.turn = enemy; state.winner = winner; state.reason = reason;
  return { captured: captured || null, winner, reason };
}

const CameraRig = {
  theta: Math.PI, phi: 0.95, radius: 11,       // start behind Red side
  _t: Math.PI, _p: 0.95, _r: 11, _shake: 0,
  update(dt) {
    this._t += (this.theta - this._t) * Math.min(1, dt * 8);
    this._p += (this.phi - this._p) * Math.min(1, dt * 8);
    this._r += (this.radius - this._r) * Math.min(1, dt * 8);
    this._shake = Math.max(0, this._shake - dt * 3);
    const s = this._shake * 0.15;
    camera.position.set(
      Math.sin(this._t) * Math.sin(this._p) * this._r + (Math.random()-0.5)*s,
      Math.cos(this._p) * this._r + (Math.random()-0.5)*s,
      Math.cos(this._t) * Math.sin(this._p) * this._r + (Math.random()-0.5)*s
    );
    camera.lookAt(0, 0, 0);
  },
  shake(strength) { this._shake = strength; }
};
